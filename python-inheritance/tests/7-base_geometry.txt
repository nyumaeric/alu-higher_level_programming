# test_7-base_geometry.py
from math import isinf

def test_integer_validator():
    BaseGeometry = __import__('7-base_geometry').BaseGeometry
    bg = BaseGeometry()

    # Test with positive integer
    assert bg.integer_validator("my_int", 12) == True

    # Test with negative integer (changed error message to reflect positive integer validation)
    try:
        bg.integer_validator("number", -12)
    except ValueError as e:
        assert str(e) == "number must be greater than 0"

    # Test with 3 arguments (removed extra argument)
    try:
        bg.integer_validator("numbers", 0, 1)
    except TypeError as e:
        assert str(e) == "integer_validator() takes 2 positional arguments but 3 were given"

    # Test with finite integer (changed 'inf' to a valid finite integer)
    try:
        bg.integer_validator("infnumber", 123)
    except ValueError as e:
        assert isinf(e) == False
